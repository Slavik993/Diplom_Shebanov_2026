```csharp
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using TMPro; // For TextMeshPro text components

public class FirstDayDialogue : MonoBehaviour
{
    public GameObject dialoguePanel; // Drag and drop the Canvas Group or RectTransform of your dialogue panel in Unity
    public TMP_Text nameText, dialogueText;
    public string[] lines = new string[10]; // Add more lines if needed

    private int currentLine = 0;

    void Start()
    {
        ShowDialogue("Привет! Я Алексей. Рад тебя видеть на нашем курсе.");
    }

    public void ShowDialogue(string name)
    {
        nameText.text = name;
        dialoguePanel.SetActive(true);
        StartCoroutine(TypeDialogue());
    }

    IEnumerator TypeDialogue()
    {
        foreach (char letter in lines[currentLine].ToCharArray())
        {
            dialogueText.text += letter;
            yield return new WaitForSeconds(0.05f);
        }

        currentLine++;

        if (currentLine < lines.Length)
        {
            StartCoroutine(TypeDialogueDelay());
        }
        else
        {
            dialoguePanel.SetActive(false);
        }
    }

    IEnumerator TypeDialogueDelay()
    {
        yield return new WaitForSeconds(3f);
        ShowDialogue("Здравствуйте, я Анна. Рад видеть тебя на нашем курсе.");
    }
}
```

Replace the lines in the `lines` array with your dialogue text for both characters:
