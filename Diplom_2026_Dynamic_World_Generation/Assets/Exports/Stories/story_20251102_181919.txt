```csharp
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using TMPro; // For TextMeshPro text components

public class RussianCultureQuest : MonoBehaviour
{
    public GameObject student; // The student NPC
    public GameObject friendNPC; // The friend NPC
    public TextMeshProUGUI studentDialogueText;
    public TextMeshProUGUI friendDialogueText;

    private string[] studentLines = new string[10];
    private string[] friendLines = new string[10];

    void Start()
    {
        // Set up the dialogues for the NPCs
        InitializeDialogues();
    }

    void InitializeDialogues()
    {
        // Student's lines
        studentLines[0] = "Здравствуйте, я столкнулся с проблемой в задаче.";
        studentLines[1] = "Может ли вы помочь мне разобраться?";
        studentLines[2] = "Я не могу понять, как решить эту задачу.";
        studentLines[3] = "Помогите, пожалуйста!";
        studentLines[4] = "Это важная часть нашей совместной работы.";
        studentLines[5] = "Я не хочу, чтобы мы отставали.";
        studentLines[6] = "Помогите мне, пожалуйста!";
        studentLines[7] = "Это важно для нас как группы.";
        studentLines[8] = "Может ли вы подсказать мне что-нибудь?";
        studentLines[9] = "Я не могу продолжить, пока не решу эту задачу.";

        // Friend NPC's lines
        friendLines[0] = "Пожалуйста, я с радостью помогу тебе.";
        friendLines[1] = "Что именно вам нужно?";
        friendLines[2] = "Да, конечно, я попышу помочь.";
        friendLines[3] = "Не беспокоитесь, я помогу.";
        friendLines[4] = "Рад, что могу помочь тебе в этой задаче.";
        friendLines[5] = "Да, конечно, мы будем работать вместе.";
        friendLines[6] = "Не беспокойтесь, я помогу тебе решить эту проблему.";
        friendLines[7] = "Мы всегда вместе.";
        friendLines[8] = "Покажи мне, что ты имеешь в виду.";
        friendLines[9] = "Рад, что могу помочь тебе.";

        // Set up the dialogue for the first interaction
        studentDialogueText.text = studentLines[0];
        friendDialogueText.text = "";
    }

    public void NextStudentLine()
    {
        if (studentLines.Length > 1)
        {
            studentDialogueText.text = studentLines[studentArrayIndex++];
            friendDialogueText.text = friendLines[friendArrayIndex % friendLines.Length];
        }
    }

    private int studentArrayIndex = 0;
    private int friendArrayIndex = 0;
}
```

This script creates a dialogue between the student and the friend NPC when the student asks for help with a task in Russian culture studies. The dialogues are set up as arrays, and the `NextStudentLine()` function is used to display the next line of dialogue from each NPC. The `InitializeDialogues()` function sets up the initial lines of dialogue for both NPCs when the scene starts.
